1. Build the image locally:
   docker build -t {{ .Values.image.repository }}:{{ .Values.image.tag }} .

2. Install or upgrade the release:
   helm upgrade --install {{ .Release.Name }} {{ .Chart.Name | printf "./helm/%s" }} \
     --set image.repository={{ .Values.image.repository }} \
     --set image.tag={{ .Values.image.tag }}

3. Reach the service:
   kubectl port-forward svc/{{ include "grove-kc.fullname" . }} 8080:{{ .Values.service.port }}
